### For loop
```cs
for (statement 1; statement 2; statement 3) 
{
  // code block to be executed
}
```
### foreach Loop
```cs
foreach (type variableName in arrayName) 
{
  // code block to be executed
}
```

### Switch
```cs
switch(expression) 
{
  case x:
    // code block
    break;
  case y:
    // code block
    break;
  default:
    // code block
    break;
}
```

### while loop
```cs
int i = 0; // initialization

while (i < 10) // condition
{
    Console.WriteLine("i = {0}", i);

    i++; // increment
}
```
